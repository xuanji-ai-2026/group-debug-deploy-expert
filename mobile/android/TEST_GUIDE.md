```powershell

# 测试所有设备
.\Run-MultiDeviceTest.ps1

# 测试指定设备
.\Run-MultiDeviceTest.ps1 -DeviceName "Pixel"

# 查看帮助
Get-Help .\Run-MultiDeviceTest.ps1 -Full
```

```bash
# 只运行单元测试

# 运行所有Instrumented Tests

./gradlew connectedAndroidTest -Pandroid.testInstrumentationRunnerArguments.class=com.beijixing.app.ExampleInstrumentedTest
```
